
```javascript
import { factorize } from './math/factorize.js';

export default async function main() {
    const input = 2 * 3 * 5 * 7 * 11 * 13;
    try {
        const primes = await factorize(input);
        console.log(`${input} = ${primes.join(' * ')}`);
    } catch (error) {
        logError(error);
    }
}
```
